Select bc.*,bcp.value,ps.plot_size from bill_charges bc left join bill_charges_replation_to_plot_size bcp ON bc.charge_id = bcp.charge_id left join plot_size ps On bcp.plot_size_id = ps.plot_size_id where (bc.charge_id = 1 OR bc.charge_id = 2 OR bc.charge_id = 3) AND (ps.plot_size_id = 1)
Select  bmi.bill_header,bmi.bill_description,bmi.bill_bank_name,bmi.bill_bank_address,bmi.bill_bank_account, CONCAT(emp.firstName,CONCAT(' ',emp.lastName)) as customer_name,ps.plot_size,btm.bill_charges_list, bcm.total_bill,bcm.amount_paid,bcm.bill_payed_date,bcm.arrears,bcm.surcharge_applied,bcm.status,btm.bill_issue_date,btm.bill_due_date from bill_consumer_month as bcm left join employee as emp ON bcm.emp_id=emp.id left Join plot_size ps ON emp.plot_id = ps.plot_size_id left Join bill_template_month as btm ON bcm.btm_id = btm.btm_id left join bill_main_info as bmi ON btm.bill_id = bmi.bill_info_id
